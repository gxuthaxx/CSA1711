from collections import deque

goal = [1,2,3,4,5,6,7,8,0]

start = [5,0,4,
         6,1,8,
         7,3,2]

def print_board(s):
    for i in range(0,9,3):
        print(s[i:i+3])
    print()

def get_neighbors(s):
    i = s.index(0)
    r, c = i//3, i%3
    moves = [(-1,0),(1,0),(0,-1),(0,1)]
    result = []

    for dr, dc in moves:
        nr, nc = r+dr, c+dc
        if 0<=nr<3 and 0<=nc<3:
            ni = nr*3 + nc
            new = s[:]
            new[i], new[ni] = new[ni], new[i]
            result.append(new)

    return result

def bfs(start):
    q = deque([(start, [])])
    visited = set([tuple(start)])

    while q:
        curr, path = q.popleft()

        if curr == goal:
            return path + [curr]

        for n in get_neighbors(curr):
            if tuple(n) not in visited:
                visited.add(tuple(n))
                q.append((n, path + [curr]))

    return None

# Run
sol = bfs(start)

if sol:
    for i, s in enumerate(sol):
        print("Step", i)
        print_board(s)
else:
    print("No solution")
